마리아 DB실행(비밀번호입력)

create database my_site;

show databases;

use mysql

create user 'jsp_user'@'127.0.0.1' identified by 'qwer1234';  // user 계정 생성

grant select,insert,update,delete,create,drop on my_site.* to 'jsp_user'@'127.0.0.1'; //권한 설정
